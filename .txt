#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <windows.h>

int main(void)
{
	//演示多个字符从两端移动，向中间汇聚
	char arr1[] = "Welcome to be BeiJing!";
	char arr2[] = "######################";
	int left = 0;
	int right = strlen(arr1)-1;
	while (left <= right)
	{
		printf("%s\n", arr2);
		arr2[left] = arr1[left];
		arr2[right] = arr1[right];
		Sleep(1000);//停顿一秒
		system("cls");//执行系统命令的一个函数-cls -清空屏幕
		left++;
		right--;
		
	}
	printf("%s\n", arr2);
	return 0;
}
